#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
#include <ctype.h>

#define MAX_HISTORY 50

typedef struct {
    char expression[100];
    double result;
} HistoryEntry;

HistoryEntry history[MAX_HISTORY];
int history_count = 0;

double add(double a, double b) {
    return a + b;
}

double subtract(double a, double b) {
    return a - b;
}

double multiply(double a, double b) {
    return a * b;
}

double divide(double a, double b) {
    if (b == 0) {
        printf("Error: Division by zero!\n");
        return 0;
    }
    return a / b;
}

double power(double a, double b) {
    return pow(a, b);
}

double modulus(double a, double b) {
    if (b == 0) {
        printf("Error: Modulus by zero!\n");
        return 0;
    }
    return fmod(a, b);
}

double factorial(double n) {
    if (n < 0) {
        printf("Error: Factorial of negative number!\n");
        return 0;
    }
    if (n == 0 || n == 1) {
        return 1;
    }
    double result = 1;
    for (int i = 2; i <= (int)n; i++) {
        result *= i;
    }
    return result;
}

double calculate_expression(const char *expr) {
    double num1, num2, result;
    char op;
    int parsed = sscanf(expr, "%lf %c %lf", &num1, &op, &num2);
    
    if (parsed != 3) {
        printf("Error: Invalid expression format!\n");
        return 0;
    }
    
    switch (op) {
        case '+':
            result = add(num1, num2);
            break;
        case '-':
            result = subtract(num1, num2);
            break;
        case '*':
            result = multiply(num1, num2);
            break;
        case '/':
            result = divide(num1, num2);
            break;
        case '^':
            result = power(num1, num2);
            break;
        case '%':
            result = modulus(num1, num2);
            break;
        default:
            printf("Error: Unknown operator '%c'!\n", op);
            return 0;
    }
    
    return result;
}

void add_to_history(const char *expr, double result) {
    if (history_count < MAX_HISTORY) {
        strcpy(history[history_count].expression, expr);
        history[history_count].result = result;
        history_count++;
    }
}

void display_history() {
    if (history_count == 0) {
        printf("\nNo calculation history.\n");
        return;
    }
    
    printf("\n=== Calculation History ===\n");
    for (int i = 0; i < history_count; i++) {
        printf("%d. %s = %.2f\n", i + 1, history[i].expression, history[i].result);
    }
    printf("\n");
}

void clear_history() {
    history_count = 0;
    printf("History cleared.\n");
}

void display_menu() {
    printf("\n╔════════════════════════════════╗\n");
    printf("║      CALCULATOR MENU           ║\n");
    printf("╠════════════════════════════════╣\n");
    printf("║ 1. Basic Calculation           ║\n");
    printf("║    (+ - * / ^ %%)              ║\n");
    printf("║ 2. Factorial (!)               ║\n");
    printf("║ 3. Square Root                 ║\n");
    printf("║ 4. Trigonometric (sin/cos/tan)║\n");
    printf("║ 5. View History                ║\n");
    printf("║ 6. Clear History               ║\n");
    printf("║ 7. Exit                        ║\n");
    printf("╚════════════════════════════════╝\n");
    printf("Enter choice (1-7): ");
}

void basic_calculation() {
    char expr[100];
    double result;
    
    printf("\nEnter expression (e.g., 10 + 5): ");
    getchar();
    fgets(expr, sizeof(expr), stdin);
    expr[strcspn(expr, "\n")] = '\0';
    
    result = calculate_expression(expr);
    printf("Result: %.2f\n", result);
    add_to_history(expr, result);
}

void factorial_calc() {
    double n;
    double result;
    
    printf("\nEnter number for factorial: ");
    scanf("%lf", &n);
    
    result = factorial(n);
    printf("%.0f! = %.0f\n", n, result);
    
    char expr[50];
    snprintf(expr, sizeof(expr), "%.0f!", n);
    add_to_history(expr, result);
}

void sqrt_calc() {
    double n;
    double result;
    
    if (n < 0) {
        printf("Error: Square root of negative number!\n");
        return;
    }
    
    printf("\nEnter number for square root: ");
    scanf("%lf", &n);
    
    if (n < 0) {
        printf("Error: Square root of negative number!\n");
        return;
    }
    
    result = sqrt(n);
    printf("√%.2f = %.2f\n", n, result);
    
    char expr[50];
    snprintf(expr, sizeof(expr), "√%.2f", n);
    add_to_history(expr, result);
}

void trig_calc() {
    int trig_choice;
    double angle;
    double result;
    char expr[50];
    
    printf("\n--- Trigonometric Functions ---\n");
    printf("1. Sine\n");
    printf("2. Cosine\n");
    printf("3. Tangent\n");
    printf("Enter choice (1-3): ");
    scanf("%d", &trig_choice);
    
    printf("Enter angle in degrees: ");
    scanf("%lf", &angle);
    
    double radians = angle * (M_PI / 180);
    
    switch (trig_choice) {
        case 1:
            result = sin(radians);
            snprintf(expr, sizeof(expr), "sin(%.2f°)", angle);
            break;
        case 2:
            result = cos(radians);
            snprintf(expr, sizeof(expr), "cos(%.2f°)", angle);
            break;
        case 3:
            result = tan(radians);
            snprintf(expr, sizeof(expr), "tan(%.2f°)", angle);
            break;
        default:
            printf("Invalid choice!\n");
            return;
    }
    
    printf("Result: %.4f\n", result);
    add_to_history(expr, result);
}

int main() {
    int choice;
    
    printf("╔════════════════════════════════╗\n");
    printf("║   Welcome to Calculator App    ║\n");
    printf("║    Made in C Programming       ║\n");
    printf("╚════════════════════════════════╝\n");
    
    while (1) {
        display_menu();
        scanf("%d", &choice);
        
        switch (choice) {
            case 1:
                basic_calculation();
                break;
            case 2:
                factorial_calc();
                break;
            case 3:
                sqrt_calc();
                break;
            case 4:
                trig_calc();
                break;
            case 5:
                display_history();
                break;
            case 6:
                clear_history();
                break;
            case 7:
                printf("\nThank you for using Calculator!\n");
                exit(0);
            default:
                printf("Invalid choice. Please try again.\n");
        }
    }
    
    return 0;
}
